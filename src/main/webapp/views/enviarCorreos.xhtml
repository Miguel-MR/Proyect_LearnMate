function toggleCheckboxes() { const container = document.getElementById("formulario:listaCorreos"); if (!container) return; const checkboxes = container.querySelectorAll("input[type='checkbox']"); const allChecked = Array.from(checkboxes).every(cb => cb.checked); checkboxes.forEach(cb => cb.checked = !allChecked); }
